#include <iostream>
using namespace std;

int inch(int, int);
double bmi(int, double);
int main(void) {
    int foot, in;
    double bang;

    cout << "请输入身高英尺、英寸、体重磅：" << endl;
    cin >> foot >> in >> bang;

    in = inch(foot, in);
    bmi(in, bang);

    return 0;
}

int inch(int foot, int inch) {
    const int F = 12;
    inch = inch + foot * F;
    cout << "您的身高是" << inch << "(英寸)\n";

    return inch;
}

double bmi(int inch, double bang) {
    double m;
    double kg;
    const double F = 0.0254;
    const double K = 2.2;

    m = inch * F;
    kg = bang / K;

    double bmi = kg / (m * m);
    cout << "您的BMI指数是：" << bmi << endl;

    return bmi;

}
```

4.秒数转换为天时分秒

```
#include <iostream>
using namespace std;

void printSec(long long);

int main(void) {
    long long sec;

    cout << "请输入整数秒：\n";
    cin >> sec;
    printSec(sec);

    return 0;
}

void printSec(long long sec) {
    const int minF = 60;
    const int hourF = 3600;
    const int dayF = hourF * 24;

    int tmp = sec;
    long long day = tmp / dayF;
    tmp %= dayF;
    long long hour = tmp / hourF;
    tmp %= hourF;
    long long min = tmp / minF;
    tmp %= minF;
    sec = tmp;

    cout << day << "days " << hour << "hours " << min << "minutes " << sec << "seconds" << endl;

}

    void setSales(Sales &s, const double arr[], int n) {
        if (n <= 0) {
            for (int i = 0; i < QUARTERS; i++) {
                s.sales[i] = 0;
                s.average = 0;
                s.max = 0;
                s.max = 0;
            } // end for
        } // end if

        double sum = arr[0];
        double max = arr[0];
        double min = arr[0];
        s.sales[0] = arr[0];
        int i = 1;
        while (i < n && i < QUARTERS) {
            s.sales[i] = arr[i];
            sum += arr[i];
            (max < arr[i]) ? max = arr[i] : false;
            (min > arr[i]) ? min = arr[i] : false;

                (min > 0) ? min = 0 : false;

        s.max = max;
